isqrtm <- function(m) {
  res <- eigen(m)
  d <- res$values
  d[d < 0] <- 0
  d[d > 0] <- 1/sqrt(d[d > 0])
  v <- res$vectors
  return (v %*% diag(d) %*% t(v))
}

Match clusters and groups.

best_trace <- 0
best_perm <- NULL
for (i in 1:10000) {
  perm <- sample(8, 8, FALSE)
  trr <- sum(diag(countmat[, perm]))
  if (trr > best_trace) {
    best_trace <- trr
    best_perm <- perm
  }
}
